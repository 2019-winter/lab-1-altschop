```python
def count_ones_with_loops(arr):
    numOfOnes = 0
    shape = arr.shape
    for i in range(shape[0]):
        for j in range(shape[1]):
            if (arr[i][j] == 1):
                numOfOnes += 1
    return numOfOnes

def count_ones_with_where(arr):
    isOne = np.where(arr == 1)
    return len(arr[isOne])
```
